# mysql
```docker
docker run -p 3306:3306 -e MYSQL_ROOT_PASSWORD=penghan123 -v /home/data/mysql/data:/var/lib/mysql:rw -v /etc/localtime:/etc/localtime:ro --name wychmod_mysql --restart=always -d mysql:8.0

# 设置连接数
show variables like '%max_connections%';

set GLOBAL max_connections=5000;

set GLOBAL mysqlx_max_connections=5000;
```


# redis

docker run -itd --name wychmod-redis1 -p 6379:6379 -v /mydata/redis/data:/data redis:6.2.4 --requirepass penghan123
