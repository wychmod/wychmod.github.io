📖 Base64 使用指南

💡 什么是 Base64？

Base64 是一种基于64个可打印字符来表示二进制数据的编码方式。它将二进制数据转换为ASCII字符串格式， 使得二进制数据可以在只支持文本的环境中传输和存储。Base64编码使用字母 A-Z、a-z、数字 0-9、符号 + 和 /， 共64个字符，以及 = 作为填充字符。

🎯 编码原理：
• 将每3个字节（24位）的二进制数据分成4组，每组6位
• 每组6位对应一个Base64字符（2^6 = 64种可能）
• 编码后的数据大小约为原数据的 4/3 倍
• 不足3字节时使用 = 进行填充

🔧 使用场景

  • 图片嵌入：在HTML/CSS中直接嵌入小图片，减少HTTP请求
  • 数据传输：在JSON、XML等文本格式中传输二进制数据
  • 邮件附件：MIME协议使用Base64编码邮件附件
  • URL安全：将二进制数据编码后安全地放入URL参数中
  • 数据存储：在文本数据库或配置文件中存储二进制数据
  • API传输：在RESTful API中传输图片等二进制数据
  • 数据加密：配合加密算法，对加密后的二进制结果进行编码

JavaScript / Node.js 
// 文本编码
const text = "Hello, Base64!";
const encoded = btoa(unescape(encodeURIComponent(text)));
console.log(encoded); // SGVsbG8sIEJhc2U2NCE=

// 文本解码
const decoded = decodeURIComponent(escape(atob(encoded)));
console.log(decoded); // Hello, Base64!

// 图片编码（浏览器环境）
const fileInput = document.querySelector('input[type="file"]');
fileInput.addEventListener('change', (e) => {
  const file = e.target.files[0];
  const reader = new FileReader();
  reader.onload = (event) => {
    const base64 = event.target.result;
    console.log(base64); // data:image/png;base64,iVBORw0KG...
  };
  reader.readAsDataURL(file);
});

// Node.js 中的 Base64
const buffer = Buffer.from("Hello, Base64!", 'utf-8');
const base64 = buffer.toString('base64');
console.log(base64); // SGVsbG8sIEJhc2U2NCE=

const decoded = Buffer.from(base64, 'base64').toString('utf-8');
console.log(decoded); // Hello, Base64!
Python 
import base64

# 文本编码
text = "Hello, Base64!"
encoded = base64.b64encode(text.encode('utf-8'))
print(encoded)  # b'SGVsbG8sIEJhc2U2NCE='

# 文本解码
decoded = base64.b64decode(encoded).decode('utf-8')
print(decoded)  # Hello, Base64!

# 图片编码
with open('image.png', 'rb') as f:
    image_data = f.read()
    encoded_image = base64.b64encode(image_data)
    print(encoded_image.decode('utf-8'))

# 图片解码
with open('decoded_image.png', 'wb') as f:
    decoded_image = base64.b64decode(encoded_image)
    f.write(decoded_image)

# URL安全的Base64编码（使用 - 和 _ 替代 + 和 /）
url_safe_encoded = base64.urlsafe_b64encode(text.encode('utf-8'))
print(url_safe_encoded)  # b'SGVsbG8sIEJhc2U2NCE='
Java 
import java.util.Base64;
import java.nio.charset.StandardCharsets;
import java.nio.file.Files;
import java.nio.file.Paths;

public class Base64Example {
    public static void main(String[] args) throws Exception {
        // 文本编码
        String text = "Hello, Base64!";
        String encoded = Base64.getEncoder()
            .encodeToString(text.getBytes(StandardCharsets.UTF_8));
        System.out.println(encoded); // SGVsbG8sIEJhc2U2NCE=

        // 文本解码
        byte[] decoded = Base64.getDecoder().decode(encoded);
        String decodedText = new String(decoded, StandardCharsets.UTF_8);
        System.out.println(decodedText); // Hello, Base64!

        // 图片编码
        byte[] imageBytes = Files.readAllBytes(Paths.get("image.png"));
        String encodedImage = Base64.getEncoder().encodeToString(imageBytes);
        System.out.println(encodedImage);

        // 图片解码
        byte[] decodedImage = Base64.getDecoder().decode(encodedImage);
        Files.write(Paths.get("decoded_image.png"), decodedImage);

        // URL安全的Base64编码
        String urlSafeEncoded = Base64.getUrlEncoder()
            .encodeToString(text.getBytes(StandardCharsets.UTF_8));
        System.out.println(urlSafeEncoded);
    }
}
Go 
package main

import (
    "encoding/base64"
    "fmt"
    "os"
)

func main() {
    // 文本编码
    text := "Hello, Base64!"
    encoded := base64.StdEncoding.EncodeToString([]byte(text))
    fmt.Println(encoded) // SGVsbG8sIEJhc2U2NCE=

    // 文本解码
    decoded, err := base64.StdEncoding.DecodeString(encoded)
    if err != nil {
        panic(err)
    }
    fmt.Println(string(decoded)) // Hello, Base64!

    // 图片编码
    imageData, _ := os.ReadFile("image.png")
    encodedImage := base64.StdEncoding.EncodeToString(imageData)
    fmt.Println(encodedImage)

    // 图片解码
    decodedImage, _ := base64.StdEncoding.DecodeString(encodedImage)
    os.WriteFile("decoded_image.png", decodedImage, 0644)

    // URL安全的Base64编码
    urlSafeEncoded := base64.URLEncoding.EncodeToString([]byte(text))
    fmt.Println(urlSafeEncoded)
}

// 检查是否为有效的Base64字符串
function isBase64($string) {
    return (bool) preg_match('/^[a-zA-Z0-9\/\r\n+]*={0,2}$/', $string);
}

⚠️ 注意事项

  • 数据膨胀：Base64编码会使数据量增加约33%，不适合大文件传输
  • 不是加密：Base64是编码方式，不是加密算法，任何人都可以解码
  • 性能考虑：编解码需要计算资源，大量数据处理时注意性能影响
  • 中文处理：编码中文前需要先进行UTF-8编码，避免乱码
  • 换行符：某些实现会在固定长度处插入换行符，解码前需要移除
  • URL传输：标准Base64中的+和/在URL中有特殊含义，需使用URL安全版本
⚠️ 安全警告：
Base64只是编码方式，不提供任何安全保护。如果需要保护敏感数据，请使用专业的加密算法（如AES、RSA等）， 然后再对加密结果进行Base64编码以便传输。

🎓 扩展知识

Base64变体：

  • 标准Base64：使用 A-Z、a-z、0-9、+、/ 和 = 填充
  • URL安全Base64：使用 - 和 _ 替代 + 和 /，避免URL编码问题
  • 无填充Base64：省略末尾的 = 填充字符，减少数据长度
  • Base64URL：专门为URL设计的变体，广泛用于JWT等场景

HTML中使用Data URI：

Data URI允许在HTML中直接嵌入Base64编码的图片，格式为： data:[<mediatype>][;base64],<data>

例如：data:image/png;base64,iVBORw0KG...

优点：减少HTTP请求，适合小图片；
缺点：增加HTML文件大小，不利于缓存，不适合大图片。
